Mapstructure 库的使用和分析

2022-08-13 2 分钟阅读 go三方库使用和源码分析go

使用viper的时候看到了这个库，觉得在项目中还是有很多使用场景的，所以拿来说一说

mapsturcture 干什么用的

根据其官网 的说明，mapstructure是用来进行通用的map和特定的结构体之间互相装换的，比如将下面的变量a转换成变量b

 1var a = map[string]interface{}{
 2    "Name": "bob",
 3    "Age":  18,
 4}
 5
 6type Person struct {
 7    Name string
 8    Age  int
 9}
10
11var b Person
12
13e := mapstructure.Decode(a, &b)
14if e != nil {
15    panic(e)
16}
17
18fmt.Printf("%#v\n", b)
19
20// 输出
21// main.Person{Name:"bob", Age:18}

为什么需要 mapsturcture

通过上面的简单演示，有些同学可能并不能很快的联想到这个库的使用场景，通常我们一般都是直接预定义好对应的结构体，然后decode就好了，比如下面的代码

 1var encodedStr = `{"name": "bob", "age": 18}`
 2type Person struct {
 3    Name string `json:"name"`
 4    Age  int    `json:"age"`
 5}
 6var p Person
 7e := json.Unmarshal([]byte(encodedStr), &p)
 8if e != nil {
 9    panic(e)
10}
11fmt.Printf("%#v\n", p)
12
13// 输出
14// main.Person{Name:"bob", Age:18}

但是，也有很多场景，我们往往在读取其中一个字段之前，不知道其他字段的具体结构，比如有一个消息队列，消息队列中的消息要根据type 字段区分不同的结构使用， 如果不使用这个类库，那我们大概率的编码方式会是，先读取type字段，然后根据type字段的类型二次解码不同的结构，代码如下：

 1	var msgs = []string{
 2		`{"type": "person", "name": "bob", "Age": 18}`,
 3		`{"type": "animal", "nickname": "wangwang"}`,
 4	}
 5
 6	type T struct {
 7		Type string `json:"type"`
 8	}
 9
10	type Person struct {
11		Name string `json:"name"`
12		Age  int    `json:"age"`
13	}
14
15	type Animal struct {
16		Nickname string `json:"nickname"`
17	}
18
19	for _, s := range msgs {
20		t := T{}
21		if e := json.Unmarshal([]byte(s), &t); e != nil {
22			panic(e)
23		} else {
24			switch t.Type {
25			case "person":
26				p := Person{}
27				_ = json.Unmarshal([]byte(s), &p)
28				fmt.Printf("%#v\n", p)
29			case "animal":
30				p := Animal{}
31				_ = json.Unmarshal([]byte(s), &p)
32				fmt.Printf("%#v\n", p)
33			default:
34				panic("not support")
35			}
36		}
37	}

如果使用了mapstructure，则可以先解析为通用的map，然后将通用的map装换成特定的结构体，代码如下

 1	var msgs = []string{
 2		`{"type": "person", "name": "bob", "Age": 18}`,
 3		`{"type": "animal", "nickname": "wangwang"}`,
 4	}
 5
 6	type Person struct {
 7		Name string `json:"name"`
 8		Age  int    `json:"age"`
 9	}
10
11	type Animal struct {
12		Nickname string `json:"nickname"`
13	}
14
15	for _, s := range msgs {
16		t := map[string]interface{}{}
17		if e := json.Unmarshal([]byte(s), &t); e != nil {
18			panic(e)
19		} else {
20			switch t["type"].(string) {
21			case "person":
22				p := Person{}
23				_ = mapstructure.Decode(t, &p)
24				fmt.Printf("%#v\n", p)
25			case "animal":
26				p := Animal{}
27				_ = mapstructure.Decode(t, &p)
28				fmt.Printf("%#v\n", p)
29			default:
30				panic("not support")
31			}
32		}
33	}
34
35	// 输出
36	//main.Person{Name:"bob", Age:18}
37	//main.Animal{Nickname:"wangwang"}
