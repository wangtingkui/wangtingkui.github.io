Mapstructure 库的使用和分析

2022-08-13 6 分钟阅读 go三方库使用和源码分析go

使用viper的时候看到了这个库，觉得在项目中还是有很多使用场景的，所以拿来说一说

mapsturcture 干什么用的

根据其官网 的说明，mapstructure是用来进行通用的map和特定的结构体之间互相装换的，比如将下面的变量a转换成变量b

 1var a = map[string]interface{}{
 2    "Name": "bob",
 3    "Age":  18,
 4}
 5
 6type Person struct {
 7    Name string
 8    Age  int
 9}
10
11var b Person
12
13e := mapstructure.Decode(a, &b)
14if e != nil {
15    panic(e)
16}
17
18fmt.Printf("%#v\n", b)
19
20// 输出
21// main.Person{Name:"bob", Age:18}

为什么需要 mapsturcture

通过上面的简单演示，有些同学可能并不能很快的联想到这个库的使用场景，通常我们一般都是直接预定义好对应的结构体，然后decode就好了，比如下面的代码

 1var encodedStr = `{"name": "bob", "age": 18}`
 2type Person struct {
 3    Name string `json:"name"`
 4    Age  int    `json:"age"`
 5}
 6var p Person
 7e := json.Unmarshal([]byte(encodedStr), &p)
 8if e != nil {
 9    panic(e)
10}
11fmt.Printf("%#v\n", p)
12
13// 输出
14// main.Person{Name:"bob", Age:18}

但是，也有很多场景，我们往往在读取其中一个字段之前，不知道其他字段的具体结构，比如有一个消息队列，消息队列中的消息要根据type 字段区分不同的结构使用， 如果不使用这个类库，那我们大概率的编码方式会是，先读取type字段，然后根据type字段的类型二次解码不同的结构，代码如下：

 1var msgs = []string{
 2	`{"type": "person", "name": "bob", "Age": 18}`,
 3	`{"type": "animal", "nickname": "wangwang"}`,
 4}
 5
 6type T struct {
 7	Type string `json:"type"`
 8}
 9
10type Person struct {
11	Name string `json:"name"`
12	Age  int    `json:"age"`
13}
14
15type Animal struct {
16	Nickname string `json:"nickname"`
17}
18
19for _, s := range msgs {
20	t := T{}
21	if e := json.Unmarshal([]byte(s), &t); e != nil {
22		panic(e)
23	} else {
24		switch t.Type {
25		case "person":
26			p := Person{}
27			_ = json.Unmarshal([]byte(s), &p)
28			fmt.Printf("%#v\n", p)
29		case "animal":
30			p := Animal{}
31			_ = json.Unmarshal([]byte(s), &p)
32			fmt.Printf("%#v\n", p)
33		default:
34			panic("not support")
35		}
36	}
37}

如果使用了mapstructure，则可以先解析为通用的map，然后将通用的map装换成特定的结构体，代码如下

 1var msgs = []string{
 2	`{"type": "person", "name": "bob", "Age": 18}`,
 3	`{"type": "animal", "nickname": "wangwang"}`,
 4}
 5
 6type Person struct {
 7	Name string `json:"name"`
 8	Age  int    `json:"age"`
 9}
10
11type Animal struct {
12	Nickname string `json:"nickname"`
13}
14
15for _, s := range msgs {
16	t := map[string]interface{}{}
17	if e := json.Unmarshal([]byte(s), &t); e != nil {
18		panic(e)
19	} else {
20		switch t["type"].(string) {
21		case "person":
22			p := Person{}
23			_ = mapstructure.Decode(t, &p)
24			fmt.Printf("%#v\n", p)
25		case "animal":
26			p := Animal{}
27			_ = mapstructure.Decode(t, &p)
28			fmt.Printf("%#v\n", p)
29		default:
30			panic("not support")
31		}
32	}
33}
34
35// 输出
36//main.Person{Name:"bob", Age:18}
37//main.Animal{Nickname:"wangwang"}

源码分析

老样子，我们还是只分析主流程，从上面例子我们可以看到，一切都可以从mapstructure.Decode方法看起，以下面的一段代码为例

 1m := map[string]interface{}{
 2	"Name": "xiaok",
 3	"Age":  18,
 4}
 5type Person struct {
 6	Name string
 7	Age  int
 8}
 9p := new(Person)
10mapstructure.Decode(m, p)
11fmt.Printf("%#v", p)

 1// Decode takes an input structure and uses reflection to translate it to
 2// the output structure. output must be a pointer to a map or struct.
 3// 很简单的一个方法，创建一个配置对象，根据配置对象生成一个Decoder（典型的建造者模式）
 4// 然后调用了Decoder的Decode方法
 5func Decode(input interface{}, output interface{}) error {
 6	// decoder的配置，会涉及钩子，如何处理字段名称等等规则，这里都用默认的
 7	config := &DecoderConfig{
 8		Metadata: nil,
 9		Result:   output,
10	}
11
12	// 生成decoder
13	decoder, err := NewDecoder(config)
14	if err != nil {
15		return err
16	}
17
18	// 调用Decode方法
19	return decoder.Decode(input)
20}

继续分析 Decoder.Decode 方法

  1// Decode decodes the given raw interface to the target pointer specified
  2// by the configuration.
  3func (d *Decoder) Decode(input interface{}) error {
  4	// 关注第三个参数，取了目的要解析到的真正类型
  5	// 比如我们传递了一个 Person 类型的指针，这里取到的就是真正的 Person 的反射值
  6	return d.decode("", input, reflect.ValueOf(d.config.Result).Elem())
  7}

  9// 直接看 decode 的实现
 10// Decodes an unknown data type into a specific reflection value.
 11func (d *Decoder) decode(name string, input interface{}, outVal reflect.Value) error {
 12	var inputVal reflect.Value
 13	if input != nil {
 14		inputVal = reflect.ValueOf(input)
 15
 16		// We need to check here if input is a typed nil. Typed nils won't
 17		// match the "input == nil" below so we check that here.
 18		// 这里主要处理 typed nil 的问题，关于 typed nil 可以参考这篇文章：https://zhuanlan.zhihu.com/p/151140497
 19		if inputVal.Kind() == reflect.Ptr && inputVal.IsNil() {
 20			input = nil
 21		}
 22	}
 23
 24	// 如果输入为空，那也没有必要解什么东西了
 25	// 这里还处理了设置了 ZeroFields 的情况，我们的主线没有设置，不关注这里了
 26	if input == nil {
 27		// If the data is nil, then we don't set anything, unless ZeroFields is set
 28		// to true.
 29		if d.config.ZeroFields {
 30			outVal.Set(reflect.Zero(outVal.Type()))
 31
 32			if d.config.Metadata != nil && name != "" {
 33				d.config.Metadata.Keys = append(d.config.Metadata.Keys, name)
 34			}
 35		}
 36		return nil
 37	}
 38
 39	// 判断inputVal是不是合法的，比如没有分配空间之类的
 40	if !inputVal.IsValid() {
 41		// If the input value is invalid, then we just set the value
 42		// to be the zero value.
 43		outVal.Set(reflect.Zero(outVal.Type()))
 44		if d.config.Metadata != nil && name != "" {
 45			d.config.Metadata.Keys = append(d.config.Metadata.Keys, name)
 46		}
 47		return nil
 48	}
 49
 50	// 钩子函数处理，不关注
 51	if d.config.DecodeHook != nil {
 52		// We have a DecodeHook, so let's pre-process the input.
 53		var err error
 54		input, err = DecodeHookExec(d.config.DecodeHook, inputVal, outVal)
 55		if err != nil {
 56			return fmt.Errorf("error decoding '%s': %s", name, err)
 57		}
 58	}
 59
 60	var err error
 61	outputKind := getKind(outVal)
 62	addMetaKey := true
 63	// 获取拟输出的类型处理，从这里也可以看到，mapstructure库不仅能够处理 map 和 structure 之间的转化
 64	// 其他类型也是能够处理的，只不过我们常用的是 map 和 structure 之间的转换罢了
 65	switch outputKind {
 66	case reflect.Bool:
 67		err = d.decodeBool(name, input, outVal)
 68	case reflect.Interface:
 69		err = d.decodeBasic(name, input, outVal)
 70	case reflect.String:
 71		err = d.decodeString(name, input, outVal)
 72	case reflect.Int:
 73		err = d.decodeInt(name, input, outVal)
 74	case reflect.Uint:
 75		err = d.decodeUint(name, input, outVal)
 76	case reflect.Float32:
 77		err = d.decodeFloat(name, input, outVal)
 78	case reflect.Struct:
 79		err = d.decodeStruct(name, input, outVal)
 80	case reflect.Map:
 81		err = d.decodeMap(name, input, outVal)
 82	case reflect.Ptr:
 83		addMetaKey, err = d.decodePtr(name, input, outVal)
 84	case reflect.Slice:
 85		err = d.decodeSlice(name, input, outVal)
 86	case reflect.Array:
 87		err = d.decodeArray(name, input, outVal)
 88	case reflect.Func:
 89		err = d.decodeFunc(name, input, outVal)
 90	default:
 91		// If we reached this point then we weren't able to decode it
 92		return fmt.Errorf("%s: unsupported type: %s", name, outputKind)
 93	}
 94
 95	// 元数据的处理
 96	// If we reached here, then we successfully decoded SOMETHING, so
 97	// mark the key as used if we're tracking metainput.
 98	if addMetaKey && d.config.Metadata != nil && name != "" {
 99		d.config.Metadata.Keys = append(d.config.Metadata.Keys, name)
100	}
101
102	return err
103}

105// 我们选择符合例子的decodeStruct进行看吧
106func (d *Decoder) decodeStruct(name string, data interface{}, val reflect.Value) error {
107	// 取源数据的直接值（也就是如果传入的是指针，取用指针指向的值）
108	dataVal := reflect.Indirect(reflect.ValueOf(data))
109
110	// If the type of the value to write to and the data match directly,
111	// then we just set it directly instead of recursing into the structure.
112	// 如果两者类型一样，直接赋值
113	if dataVal.Type() == val.Type() {
114		val.Set(dataVal)
115		return nil
116	}
117
118	// 根据源数据类型进行处理
119	dataValKind := dataVal.Kind()
120	switch dataValKind {
121	case reflect.Map:
122	// 如果源数据是 map
123		return d.decodeStructFromMap(name, dataVal, val)
124		
125	case reflect.Struct:
126	// 如果源数据是 struct
127		// Not the most efficient way to do this but we can optimize later if
128		// we want to. To convert from struct to struct we go to map first
129		// as an intermediary.
130
131		// Make a new map to hold our result
132		mapType := reflect.TypeOf((map[string]interface{})(nil))
133		mval := reflect.MakeMap(mapType)
134
135		// Creating a pointer to a map so that other methods can completely
136		// overwrite the map if need be (looking at you decodeMapFromMap). The
137		// indirection allows the underlying map to be settable (CanSet() == true)
138		// where as reflect.MakeMap returns an unsettable map.
139		addrVal := reflect.New(mval.Type())
140
141		reflect.Indirect(addrVal).Set(mval)
142		if err := d.decodeMapFromStruct(name, dataVal, reflect.Indirect(addrVal), mval); err != nil {
143			return err
144		}
145
146		result := d.decodeStructFromMap(name, reflect.Indirect(addrVal), val)
147		return result
148
149	default:
150		return fmt.Errorf("'%s' expected a map, got '%s'", name, dataVal.Kind())
151	}
152}

154// 看如何将一个 map 转成 struct
155func (d *Decoder) decodeStructFromMap(name string, dataVal, val reflect.Value) error {
156	dataValType := dataVal.Type()
157	if kind := dataValType.Key().Kind(); kind != reflect.String && kind != reflect.Interface {
158		return fmt.Errorf(
159			"'%s' needs a map with string keys, has '%s' keys",
160			name, dataValType.Key().Kind())
161	}
162
163	dataValKeys := make(map[reflect.Value]struct{})
164	dataValKeysUnused := make(map[interface{}]struct{})
165	for _, dataValKey := range dataVal.MapKeys() {
166		dataValKeys[dataValKey] = struct{}{}
167		dataValKeysUnused[dataValKey.Interface()] = struct{}{}
168	}
169
170	targetValKeysUnused := make(map[interface{}]struct{})
171	errors := make([]string, 0)
172
173	// This slice will keep track of all the structs we'll be decoding.
174	// There can be more than one struct if there are embedded structs
175	// that are squashed.
176	structs := make([]reflect.Value, 1, 5)
177	structs[0] = val
178
179	// Compile the list of all the fields that we're going to be decoding
180	// from all the structs.
181	type field struct {
182		field reflect.StructField
183		val   reflect.Value
184	}
185
186	// remainField is set to a valid field set with the "remain" tag if
187	// we are keeping track of remaining values.
188	var remainField *field
189
190	fields := []field{}
191	for len(structs) > 0 {
192		structVal := structs[0]
193		structs = structs[1:]
194
195		structType := structVal.Type()
196
197		for i := 0; i < structType.NumField(); i++ {
198			fieldType := structType.Field(i)
199			fieldVal := structVal.Field(i)
200			if fieldVal.Kind() == reflect.Ptr && fieldVal.Elem().Kind() == reflect.Struct {
201				// Handle embedded struct pointers as embedded structs.
202				fieldVal = fieldVal.Elem()
203			}
204
205			// If "squash" is specified in the tag, we squash the field down.
206			squash := d.config.Squash && fieldVal.Kind() == reflect.Struct && fieldType.Anonymous
207			remain := false
208
209			// We always parse the tags cause we're looking for other tags too
210			tagParts := strings.Split(fieldType.Tag.Get(d.config.TagName), ",")
211			for _, tag := range tagParts[1:] {
212				if tag == "squash" {
213					squash = true
214					break
215				}
216
217				if tag == "remain" {
218					remain = true
219					break
220				}
221			}
222
223			if squash {
224				if fieldVal.Kind() != reflect.Struct {
225					errors = appendErrors(errors,
226						fmt.Errorf("%s: unsupported type for squash: %s", fieldType.Name, fieldVal.Kind()))
227				} else {
228					structs = append(structs, fieldVal)
229				}
230				continue
231			}
232
233			// Build our field
234			if remain {
235				remainField = &field{fieldType, fieldVal}
236			} else {
237				// Normal struct field, store it away
238				fields = append(fields, field{fieldType, fieldVal})
239			}
240		}
241	}
242
243	// for fieldType, field := range fields {
244	for _, f := range fields {
245		field, fieldValue := f.field, f.val
246		fieldName := field.Name
247
248		tagValue := field.Tag.Get(d.config.TagName)
249		tagValue = strings.SplitN(tagValue, ",", 2)[0]
250		if tagValue != "" {
251			fieldName = tagValue
252		}
253
254		rawMapKey := reflect.ValueOf(fieldName)
255		rawMapVal := dataVal.MapIndex(rawMapKey)
256		if !rawMapVal.IsValid() {
257			// Do a slower search by iterating over each key and
258			// doing case-insensitive search.
259			for dataValKey := range dataValKeys {
260				mK, ok := dataValKey.Interface().(string)
261				if !ok {
262					// Not a string key
263					continue
264				}
265
266				if d.config.MatchName(mK, fieldName) {
267					rawMapKey = dataValKey
268					rawMapVal = dataVal.MapIndex(dataValKey)
269					break
270				}
271			}
272
273			if !rawMapVal.IsValid() {
274				// There was no matching key in the map for the value in
275				// the struct. Remember it for potential errors and metadata.
276				targetValKeysUnused[fieldName] = struct{}{}
277				continue
278			}
279		}
280
281		if !fieldValue.IsValid() {
282			// This should never happen
283			panic("field is not valid")
284		}
285
286		// If we can't set the field, then it is unexported or something,
287		// and we just continue onwards.
288		if !fieldValue.CanSet() {
289			continue
290		}
291
292		// Delete the key we're using from the unused map so we stop tracking
293		delete(dataValKeysUnused, rawMapKey.Interface())
294
295		// If the name is empty string, then we're at the root, and we
296		// don't dot-join the fields.
297		if name != "" {
298			fieldName = name + "." + fieldName
299		}
300
301		if err := d.decode(fieldName, rawMapVal.Interface(), fieldValue); err != nil {
302			errors = appendErrors(errors, err)
303		}
304	}
305
306	// If we have a "remain"-tagged field and we have unused keys then
307	// we put the unused keys directly into the remain field.
308	if remainField != nil && len(dataValKeysUnused) > 0 {
309		// Build a map of only the unused values
310		remain := map[interface{}]interface{}{}
311		for key := range dataValKeysUnused {
312			remain[key] = dataVal.MapIndex(reflect.ValueOf(key)).Interface()
313		}
314
315		// Decode it as-if we were just decoding this map onto our map.
316		if err := d.decodeMap(name, remain, remainField.val); err != nil {
317			errors = appendErrors(errors, err)
318		}
319
320		// Set the map to nil so we have none so that the next check will
321		// not error (ErrorUnused)
322		dataValKeysUnused = nil
323	}
324
325	if d.config.ErrorUnused && len(dataValKeysUnused) > 0 {
326		keys := make([]string, 0, len(dataValKeysUnused))
327		for rawKey := range dataValKeysUnused {
328			keys = append(keys, rawKey.(string))
329		}
330		sort.Strings(keys)
331
332		err := fmt.Errorf("'%s' has invalid keys: %s", name, strings.Join(keys, ", "))
333		errors = appendErrors(errors, err)
334	}
335
336	if d.config.ErrorUnset && len(targetValKeysUnused) > 0 {
337		keys := make([]string, 0, len(targetValKeysUnused))
338		for rawKey := range targetValKeysUnused {
339			keys = append(keys, rawKey.(string))
340		}
341		sort.Strings(keys)
342
343		err := fmt.Errorf("'%s' has unset fields: %s", name, strings.Join(keys, ", "))
344		errors = appendErrors(errors, err)
345	}
346
347	if len(errors) > 0 {
348		return &Error{errors}
349	}
350
351	// Add the unused keys to the list of unused keys if we're tracking metadata
352	if d.config.Metadata != nil {
353		for rawKey := range dataValKeysUnused {
354			key := rawKey.(string)
355			if name != "" {
356				key = name + "." + key
357			}
358
359			d.config.Metadata.Unused = append(d.config.Metadata.Unused, key)
360		}
361		for rawKey := range targetValKeysUnused {
362			key := rawKey.(string)
363			if name != "" {
364				key = name + "." + key
365			}
366
367			d.config.Metadata.Unset = append(d.config.Metadata.Unset, key)
368		}
369	}
370
371	return nil
372}
