Go进行文件操作

Jan 05, 2020
阅读时长: 2 分钟

go 的 os 包提供了与基础的与操作系统无关的文件操作，包括创建、删除、读写等。同时为了更便捷的操作和性能上的提升，go在 io/ioutil 、bufio 等包中也提供了一些常用的方法供开发者使用

本文只列出了一些常用的方法，更多的操作可以参考：https://colobu.com/2016/10/12/go-file-operations

使用 os 包进行文件操作

打开文件

package main

import (
	"os"
)

func main() {

	var (
		file *os.File
		err  error
	)

	// 以只读方式打开一个文件
	file, err = os.Open("a.txt")
	if err != nil {
		panic(err)
	}
	_ = file.Close()

	// 以指定的模式打开文件，这个方法更具有通用性
	// 一般情况下我们应该尽量使用 os.Open 和 os.Create 方法而不是这个方法
	file, err = os.OpenFile("a.txt", os.O_APPEND, 0666)
	if err != nil {
		panic(err)
	}
	_ = file.Close()
}

创建文件

package main

import (
	"os"
)

func main() {

	// 创建一个空文件
	// 看源码就可以发现它其实就是 OpenFile(name, O_RDWR|O_CREATE|O_TRUNC, 0666) 的快捷方式
	file, err := os.Create("create.txt")
	if err != nil {
		panic(err)
	}
	_ = file.Close()

	// 使用给定的unix文件描述符和名称创建文件，一般不使用
	file = os.NewFile(3, "haha.txt")
	_ = file.Close()
}

删除文件

package main

import (
	"os"
)

func main() {
	err := os.Remove("create.txt")
	if err != nil {
		panic(err)
	}
}

重命名和移动文件

package main

import (
	"os"
)

func main() {
	err := os.Rename("a.txt", "b.txt")
	if err != nil {
		panic(err)
	}
}

清空文件

package main

import (
	"os"
)

func main() {
	// 第二个参数是要保留的字节数，如果传入0，就是清空整个文件
	err := os.Truncate("b.txt", 2)
	if err != nil {
		panic(err)
	}
}

获取文件信息

package main

import (
	"fmt"
	"os"
)

var (
	fileInfo os.FileInfo
	err      error
)

func main() {
	fileInfo, err = os.Stat("b.txt")
	if err != nil {
		panic(err)
	}
	fmt.Println("File name:", fileInfo.Name())
	fmt.Println("Size in bytes:", fileInfo.Size())
	fmt.Println("Permissions:", fileInfo.Mode())
	fmt.Println("Last modified:", fileInfo.ModTime())
	fmt.Println("Is Directory: ", fileInfo.IsDir())
	fmt.Printf("System interface type: %T\n", fileInfo.Sys())
	fmt.Printf("System info: %+v\n\n", fileInfo.Sys())
}

文件读取

package main

import (
	"fmt"
	"os"
)

func main() {
	file, err := os.Open("b.txt")
	if err != nil {
		panic(err)
	}

	buf := make([]byte, 10)
	for {
		l, err := file.Read(buf)
		if l > 0 {
			fmt.Println(string(buf[:l]))
		}
		if err != nil {
			break
		}
	}
}

写文件

package main

import (
	"os"
)

func main() {
	file, err := os.Create("newfile.txt")
	if err != nil {
		panic(err)
	}
	defer file.Close()

	file.Write([]byte("fuck1"))
	file.WriteString("fuck2")
}

使用 bufio 操作文件

按行读取文件

package main

import (
	"bufio"
	"fmt"
	"os"
)

func main() {
	file, err := os.Open("b.txt")
	if err != nil {
		panic(err)
	}
	defer file.Close()

	// 通过 bufio.Reader 读取行
	// 默认 buffer 是4096
	reader := bufio.NewReader(file)
	for {
		// 读取文件的一行，默认用 \r\n 或者 \n 分隔，读取出来的行是不包含分隔符的
		// 如果文件的行太长，h超过了 bufio.Reader 内部的 buffer， 会将第二个返回参数 isPrefix 设置为 true
		buf, _, err := reader.ReadLine()
		if err != nil {
			break
		}
		fmt.Println(string(buf))
	}

	// 通过bufio.Scanner 读取行
	file.Seek(0, 0)
	scanner := bufio.NewScanner(file)
	// scanner.Split() ，可以通过这个函数设置自定义的分隔方式，默认是换行
	for scanner.Scan() {
		fmt.Println(scanner.Text())
	}
}
