Go标准库（net/Url）

2019-12-21 3 分钟阅读 go标准库使用和源码分析go

net/url包是官方为我们提供的处理 url 链接的库，基本能覆盖我们大多数场景下的使用

url 结构

想要使用好这个库，我们需要了解url的基本结构：标准结构：scheme://[userinfo@]host/path[?query][#fragment]，如果 scheme 之后不是以斜线开头的，那么结构被认为是 scheme:opaque[?query][#fragment]，也就是用户信息、域名和路径部分是不透明的，可以认为是被加密或者被编码过的

更多详细的信息，可以参考百科

环境

1❯ go version               
2go version go1.16.15 darwin/amd64

包中的基本结构体

先概览下包中导出的方法和结构

可以看到包中有3个比较主要的结构：

  • URL：代表一个被解析的url
  • UserInfo：代表 url 中的用户信息部分
  • Values：存放 query 参数或者 form 参数
 1type URL struct {
 2	Scheme      string
 3	Opaque      string    // encoded opaque data
 4	User        *Userinfo // username and password information
 5	Host        string    // host or host:port
 6	Path        string    // path (relative paths may omit leading slash)
 7	RawPath     string    // encoded path hint (see EscapedPath method)
 8	ForceQuery  bool      // append a query ('?') even if RawQuery is empty
 9	RawQuery    string    // encoded query values, without '?'
10	Fragment    string    // fragment for references, without '#'
11	RawFragment string    // encoded fragment hint (see EscapedFragment method)
12}
13
14type Userinfo struct {
15	username    string
16	password    string
17	passwordSet bool
18}
19
20type Values map[string][]string

基本使用

解析url

 1func main() {
 2	rowUrl := "http://xiaok:mypwd@xiaokspace.com:8080/mypath/xxx?a=123&b=456&c=中文#tag"
 3	u, err := url.Parse(rowUrl)
 4	if err != nil {
 5		panic(err)
 6	}
 7	fmt.Println(u.Scheme)      // http
 8	fmt.Println(u.Opaque)      //
 9	fmt.Println(u.User)        // xiaok:mypwd
10	fmt.Println(u.Host)        // xiaokspace.com:8080
11	fmt.Println(u.Path)        // /mypath/xxx
12	fmt.Println(u.RawPath)     //
13	fmt.Println(u.RawQuery)    // a=123&b=456&c=中文
14	fmt.Println(u.Fragment)    // tag
15	fmt.Println(u.RawFragment) //
16
17	// 不过有的情况也不会直接使用暴露的值，而是通过url中封装好的方法使用
18	fmt.Println(u.EscapedPath())     // /mypath/xxx
19	fmt.Println(u.EscapedFragment()) // tag
20	fmt.Println(u.Hostname())        // xiaokspace.com
21	fmt.Println(u.Port())            // 8080
22	fmt.Println(u.RequestURI())      // /mypath/xxx?a=123&b=456&c=中文
23}

在 url 包中，有三个关于解析的方法，Parse,ParseRequestURI和ParseQuery：

  • Parse 解析一个通用的 url（url不仅仅代表http中的地址）
  • ParseRequestURI 断定入参是从正常的 http request 中来的
  • ParseQuery 解析 query 的参数

构造url

 1func main() {
 2	u := url.URL{
 3		Scheme:   "http",
 4		Host:     "baidu.com",
 5		Path:     "/s",
 6		RawQuery: "a=c",
 7		Fragment: "tag",
 8	}
 9	fmt.Println(u.String()) //http://baidu.com/s?a=c#tag
10}

编码和解码操作

 1func main() {
 2
 3	var (
 4		rowUrl       = "http://www.baidu.com?q=我 曹#123"
 5		escapedUrl   string
 6		unescapedUrl string
 7	)
 8
 9	// 可以保证安全的放在query中
10	escapedUrl = url.QueryEscape(rowUrl)
11	unescapedUrl, _ = url.QueryUnescape(escapedUrl)
12
13	fmt.Println(escapedUrl)
14	fmt.Println(unescapedUrl)
15
16	// 可以保证安全的放在path段中
17	escapedUrl = url.PathEscape(rowUrl)
18	unescapedUrl, _ = url.PathUnescape(escapedUrl)
19
20	fmt.Println(escapedUrl)
21	fmt.Println(unescapedUrl)
22}
23
24// http%3A%2F%2Fwww.baidu.com%3Fq%3D%E6%88%91+%E6%9B%B9%23123
25// http://www.baidu.com?q=我 曹#123
26// http:%2F%2Fwww.baidu.com%3Fq=%E6%88%91%20%E6%9B%B9%23123
27// http://www.baidu.com?q=我 曹#123

URL 其他常用方法

1func main() {
2
3	rowUrl := "http://www.baidu.com/fuck?a=a&b=b&c=c#123"
4	url, _ := url.Parse(rowUrl)
5
6	// 是否是绝对路径（scheme是否为空）
7	fmt.Println(url.IsAbs()) // true
8}

Values 常用方法

 1func main() {
 2
 3	rowUrl := "http://www.baidu.com/fuck?a=a&b=b&c=c#123"
 4	url, _ := url.Parse(rowUrl)
 5
 6	values := url.Query()
 7	fmt.Println(values)
 8
 9	// 添加一个元素
10	values.Add("d", "d")
11	fmt.Println(values)
12
13	// 删除一个元素
14	values.Del("a")
15	fmt.Println(values)
16
17	// 获取一个元素
18	fmt.Println(values.Get("b"))
19
20	// 设置一个元素
21	values.Set("c", "ccc")
22	fmt.Println(values)
23	values.Set("name", "王")
24
25	// 编码成 url query 格式
26	fmt.Println(values.Encode())
27}
28
29// map[a:[a] b:[b] c:[c]]
30// map[a:[a] b:[b] c:[c] d:[d]]
31// map[b:[b] c:[c] d:[d]]
32// b
33// map[b:[b] c:[ccc] d:[d]]
34// b=b&c=ccc&d=d&name=%E7%8E%8B
